export const CurrencyPayoutRoutesDropdown = () => {
  const currencyData = [
    {
      value: '',
      label: 'Select a Currency...',
      details: {
        message: 'Please choose a currency to see its available payout routes.'
      },
      disabled: true
    },
    {
      value: 'USD',
      label: 'US Dollar (USD)',
      details: {
        international: [
          { id: 'usd-swift', name: 'SWIFT', cutoff: '14:00 UTC', deliveryTime: '1-3 business days' }
        ],
        local: [
          { id: 'usd-ach', name: 'ACH Transfer', cutoff: '16:00 EST', deliveryTime: '1-2 business days' },
          { id: 'usd-fedwire', name: 'Fedwire', cutoff: '17:00 EST', deliveryTime: 'Same day' }
        ]
      }
    },
    {
      value: 'EUR',
      label: 'Euro (EUR)',
      details: {
        international: [
          { id: 'eur-swift', name: 'SWIFT', cutoff: '15:00 CET', deliveryTime: '1-2 business days' }
        ],
        local: [
          { id: 'eur-sepa', name: 'SEPA Credit Transfer', cutoff: '16:00 CET', deliveryTime: 'Same day / Next day' },
          { id: 'eur-target2', name: 'TARGET2', cutoff: '14:00 CET', deliveryTime: 'Same day (often real-time)' }
        ]
      }
    },
    {
      value: 'GBP',
      label: 'British Pound (GBP)',
      details: {
        international: [
          { id: 'gbp-swift', name: 'SWIFT', cutoff: '15:00 GMT', deliveryTime: '1-2 business days' }
        ],
        local: [
          { id: 'gbp-bacs', name: 'BACS', cutoff: 'End of day (varies)', deliveryTime: '2-3 business days' },
          { id: 'gbp-chaps', name: 'CHAPS', cutoff: '15:30 GMT', deliveryTime: 'Same day' },
          { id: 'gbp-fps', name: 'Faster Payments', cutoff: 'Near real-time', deliveryTime: 'Near real-time (usually within minutes)' }
        ]
      }
    },
    {
      value: 'AUD',
      label: 'Australian Dollar (AUD)',
      details: {
        international: [
          { id: 'aud-swift', name: 'SWIFT', cutoff: '14:00 AEST', deliveryTime: '1-3 business days' }
        ],
        local: [
          { id: 'aud-becs', name: 'BECS (Direct Entry)', cutoff: '18:00 AEST', deliveryTime: '1-2 business days' }
        ]
      }
    },
    {
      value: 'JPY',
      label: 'Japanese Yen (JPY)',
      details: {
        international: [
          { id: 'jpy-swift', name: 'SWIFT', cutoff: '14:00 JST', deliveryTime: '1-3 business days' }
        ],
        local: [] // Example: No specific local routes listed here, or they are covered by SWIFT.
      }
    }
  ];

  const [selectedValue, setSelectedValue] = useState(currencyData[0].value);

  const handleChange = (event) => {
    setSelectedValue(event.target.value);
  };

  const selectedOption = currencyData.find(opt => opt.value === selectedValue);

  // Helper function to render a list of routes
  const renderRoutes = (routes, type) => {
    if (!routes || routes.length === 0) {
      return <p>No specific {type.toLowerCase()} routes listed for this currency.</p>;
    }
    return (
      <ul style={{ listStyleType: 'none', paddingLeft: '0' }}>
        {routes.map(route => (
          <li key={route.id} style={{ border: '1px solid #eee', padding: '10px', marginBottom: '10px', borderRadius: '4px' }}>
            <strong style={{ display: 'block', marginBottom: '5px', color: '#007bff' }}>{route.name}</strong>
            <div><strong>Cutoff Time:</strong> {route.cutoff}</div>
            <div><strong>Delivery Time:</strong> {route.deliveryTime}</div>
          </li>
        ))}
      </ul>
    );
  };

  return (
    <div style={{ border: '1px solid #e0e0e0', padding: '20px', borderRadius: '8px', marginBottom: '25px', backgroundColor: '#fdfdfd' }}>
      <label htmlFor="currency-routes-dropdown" style={{ marginRight: '10px', fontWeight: '600', display: 'block', marginBottom: '8px' }}>
        Select Currency for Payout Route Information:
      </label>
      <select
        id="currency-routes-dropdown"
        value={selectedValue}
        onChange={handleChange}
        style={{ padding: '10px', borderRadius: '4px', border: '1px solid #ccc', width: '100%', boxSizing: 'border-box', marginBottom: '20px' }}
      >
        {currencyData.map((option) => (
          <option key={option.value} value={option.value} disabled={option.disabled}>
            {option.label}
          </option>
        ))}
      </select>

      {selectedOption && selectedOption.details && (
        <div style={{ marginTop: '10px' }}>
          {selectedOption.details.message ? (
            <p>{selectedOption.details.message}</p>
          ) : (
            <>
              <h4 style={{ marginTop: '0', marginBottom: '10px', color: '#333' }}>
                Payout Routes for {selectedOption.label}:
              </h4>
              <div>
                <h5 style={{ marginBottom: '8px', borderBottom: '1px solid #ddd', paddingBottom: '5px' }}>International Routes:</h5>
                {renderRoutes(selectedOption.details.international, 'International')}
              </div>
              <div style={{ marginTop: '20px' }}>
                <h5 style={{ marginBottom: '8px', borderBottom: '1px solid #ddd', paddingBottom: '5px' }}>Local Routes:</h5>
                {renderRoutes(selectedOption.details.local, 'Local')}
              </div>
            </>
          )}
        </div>
      )}
    </div>
  );
};


<CurrencyPayoutRoutesDropdown />
