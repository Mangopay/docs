  
```javascript NodeJS  
const mangopayInstance = require('mangopay2-nodejs-sdk')
const mangopay = new mangopayInstance({
  clientId: 'your-client-id',
  clientApiKey: 'your-api-key',
})

let kycDocument = {
  Id: 'kyc_01HSXNH060XEFTTQ57RKYBHHQ1',
}

const getKycDocument = async (kycDocumentId) => {
  return await mangopay.KycDocuments.get(kycDocumentId)
    .then((response) => {
      console.info(response)
      return response
    })
    .catch((err) => {
      console.log(err)
      return false
    })
}

getKycDocument(kycDocument.Id)
```
