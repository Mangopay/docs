using MangoPay.SDK;
using MangoPay.SDK.Entities;
using Newtonsoft.Json;

class Program
{
    static async Task Main(string[] args)
    {
        MangoPayApi api = new MangoPayApi();

        api.Config.ClientId = "your-client-id";
        api.Config.ClientPassword = "your-api-key";

        var clientWalletId = "FEES_EUR";

        var viewTransactions = await api.Wallets.GetTransactionsAsync(clientWalletId, new Pagination(1, 100), null);

        string prettyPrint = JsonConvert.SerializeObject(viewTransactions, Formatting.Indented);
        Console.WriteLine(prettyPrint);
	}
}
