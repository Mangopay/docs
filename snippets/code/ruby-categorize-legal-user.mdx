require 'mangopay'

MangoPay.configure do |client|
  client.preproduction = true
  client.client_id = 'your-mangopay-client-id'
  client.client_apiKey = 'your-mangopay-api-key'
  client.log_file = File.join(Dir.pwd, 'mangopay.log')
end

def categorizeLegalUser(userId, params)
  begin
    response = MangoPay::LegalUserSca.categorize(userId, params)
    puts response
    return response
  rescue MangoPay::ResponseError => error
    puts "Failed to categorize User: #{error.message}"
    puts "Error details: #{error.details}"
    return false
  end
end
