using MangoPay.SDK;
using MangoPay.SDK.Entities.POST;
using Newtonsoft.Json;

class Program
{
    static async Task Main(string[] args)
    {
        MangoPayApi api = new MangoPayApi();

        api.Config.ClientId = "your-client-id";
        api.Config.ClientPassword = "your-api-key";

        var quoteId = "cvrquote_01J53JS92WDW6D0VY5136YTGJ7";
        var userId = "user_m_01J2TZ261WZNDM0ZDRWGDYA4GN";
        var debitedWalletId = "wlt_m_01J30991BXBB7VF28PBS82EWD3";
        var creditedWalletId = "wlt_m_01J2Y1732X0WME2FNQJQ25KC19";

        var quotedConversion = new QuotedConversionPostDTO(
            quoteId, userId,
            debitedWalletId, creditedWalletId,
            "Created using the Mangopay .NET SDK"
        );

        var createQuotedConversion = await api.Conversions.CreateQuotedConversion(quotedConversion);

        string prettyPrint = JsonConvert.SerializeObject(createQuotedConversion, Formatting.Indented);
        Console.WriteLine(prettyPrint);
	}
}
