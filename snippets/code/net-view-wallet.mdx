using MangoPay.SDK;
using MangoPay.SDK.Core;
using Newtonsoft.Json;

public class ViewWalletSca
{
    public void Run()
    {
        Task.Run(async () =>
        {
            MangoPayApi api = new MangoPayApi();

            api.Config.ClientId = "your-client-id";
            api.Config.ClientPassword = "your-api-key";

            var walletId = "wlt_m_01JYEJCS18438SEW2773PSNWCV";

            try
            {
                await api.Wallets.GetAsync(walletId, "USER_PRESENT");
            }
            catch (ResponseException ex)
            {
                Dictionary<string, string> data = ex.ResponseError.Data;
                string prettyPrint = JsonConvert.SerializeObject(data, Formatting.Indented);
                Console.WriteLine(prettyPrint);
            }
        }).GetAwaiter().GetResult();
    }
}
