function CurrencyPayoutRoutes() {
  // Get the list of currency codes from the data
  const availableCurrencies = Object.keys(payoutRoutesData);

  // State to store the currently selected currency.
  // Initialize with the first currency in the list, or null if no currencies.
  const [selectedCurrency, setSelectedCurrency] = useState(
    availableCurrencies.length > 0 ? availableCurrencies[0] : null
  );

  // Handler function to update selectedCurrency when the dropdown changes.
  const handleCurrencyChange = (event) => {
    setSelectedCurrency(event.target.value);
  };

  // Get the routes for the currently selected currency.
  const currentRoutes = selectedCurrency ? payoutRoutesData[selectedCurrency] : null;

  if (!availableCurrencies.length) {
    return <p>No currency data available.</p>;
  }

  return (
    <div style={{ fontFamily: 'Arial, sans-serif', border: '1px solid #eee', padding: '20px', borderRadius: '8px' }}>
      <h3 style={{ marginTop: '0' }}>Select Currency for Payout Routes</h3>
      <div style={{ marginBottom: '15px' }}>
        <label htmlFor="currency-select" style={{ marginRight: '10px', fontWeight: 'bold' }}>Currency:</label>
        <select
          id="currency-select"
          value={selectedCurrency}
          onChange={handleCurrencyChange}
          style={{ padding: '8px', borderRadius: '4px', border: '1px solid #ccc' }}
        >
          {availableCurrencies.map((currency) => (
            <option key={currency} value={currency}>
              {currency}
            </option>
          ))}
        </select>
      </div>

      {selectedCurrency && currentRoutes && (
        <div>
          <h4>Payout Routes for {selectedCurrency}:</h4>
          {currentRoutes.international && currentRoutes.international.length > 0 && (
            <div>
              <strong>International:</strong>
              <ul style={{ listStyleType: 'disc', paddingLeft: '20px' }}>
                {currentRoutes.international.map((route, index) => (
                  <li key={`int-${index}`}>{route}</li>
                ))}
              </ul>
            </div>
          )}
          {currentRoutes.local && currentRoutes.local.length > 0 ? (
            <div>
              <strong>Local:</strong>
              <ul style={{ listStyleType: 'disc', paddingLeft: '20px' }}>
                {currentRoutes.local.map((route, index) => (
                  <li key={`loc-${index}`}>{route}</li>
                ))}
              </ul>
            </div>
          ) : (
            currentRoutes.local !== undefined && <p><em>No specific local routes available for {selectedCurrency}.</em></p>
          )}
        </div>
      )}
    </div>
  );
}

export default CurrencyPayoutRoutes;
