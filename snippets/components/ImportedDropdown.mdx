export const ImportedDropdown = () => {
  const options = [
    { value: 'option1', label: 'Choice A', customText: 'You picked Choice A. Excellent decision!' },
    { value: 'option2', label: 'Choice B', customText: 'Choice B selected. A very popular choice.' },
    { value: 'option3', label: 'Choice C', customText: 'Ah, Choice C! The adventurous one.' }
  ];

  const [selectedValue, setSelectedValue] = useState(options[0].value);

  const handleDropdownChange = (event) => {
    setSelectedValue(event.target.value);
  };

  const selectedOptionObject = options.find(opt => opt.value === selectedValue);

  return (
    <div style={{ fontFamily: 'Arial, sans-serif', border: '1px solid #f0f0f0', padding: '15px', borderRadius: '5px' }}>
      <select
        value={selectedValue}
        onChange={handleDropdownChange}
        style={{ padding: '8px', marginRight: '10px', borderRadius: '4px', minWidth: '150px' }}
      >
        {options.map(option => (
          <option key={option.value} value={option.value}>
            {option.label}
          </option>
        ))}
      </select>
      <div style={{ marginTop: '15px', padding: '10px', backgroundColor: '#f9f9f9', border: '1px dashed #ccc', minHeight: '40px' }}>
        <strong>Selected Info:</strong> {selectedOptionObject ? selectedOptionObject.customText : 'No option selected'}
      </div>
    </div>
  );
};
